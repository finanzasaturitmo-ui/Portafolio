function calc(row) {

  const cur = row.qty * row.px;

  const inv = row.qty * row.avg;

  return { ...row, cur, inv, gl: row.avg>0?(row.px-row.avg)/row.avg*100:0, pnl: cur-inv };

}

function upd(tid, id, field, val) {

  const arr = tid==='sw'?SW:XT;

  const row = arr.find(x=>x.id===id);

  if (!row) return;

  row[field] = (field==='t') ? val.toUpperCase() : r(val);

  if (field==='qty' && r(val)>0) row.p = false;

  recalc();

}

function saveData() {

  try {

    const dataObj = {

      SW, XT, 

      rate: document.getElementById('rateInput').value,

      skInv: document.getElementById('skInv').value,

      skCur: document.getElementById('skCur').value,

      ts: new Date().toISOString()

    };

    localStorage.setItem('miPortafolioPaisa', JSON.stringify(dataObj));

    document.getElementById('lastSaved').textContent = 'Guardado: ' + new Date().toLocaleTimeString('es-CO');

  } catch(e) { showToast('Error al guardar'); }

}


function loadSaved() {

  try {

    const res = localStorage.getItem('miPortafolioPaisa');

    if (res) {

      const d = JSON.parse(res);

      if (d.SW) SW = d.SW;

      if (d.XT) XT = d.XT;

      if (d.rate) document.getElementById('rateInput').value = d.rate;

      if (d.skInv) document.getElementById('skInv').value = d.skInv;

      if (d.skCur) document.getElementById('skCur').value = d.skCur;

      if (d.ts) document.getElementById('lastSaved').textContent = 'Cargado: '+new Date(d.ts).toLocaleTimeString('es-CO');

    }

  } catch(e) {}

  

  fetchRealTimeData(); 

  recalc();

}
